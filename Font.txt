 ,--------.
(  MODE-7  )
 `--------'

By current [2022] ways of thinking, Mode-7 might be thought of as
"two extended fonts", but it is a /single/ VDU "Character Set", which
(commonly) has an "Alphanumeric Mode", and a "Graphics Mode".

Mode-7 is an 8bit Character Set, so each Mode has 256 entries.
These entries are split between "Glyphs" and "Control Codes".
  # Glyphs        are     Printable Characters: {Letters, Numbers, Symbols, Graphics}
  # Control Codes are Non-Printable Characters: {Move cursor, Select colour, ...}
      and are used to control the State Machine in the VDU hardware

There Exists:
  # A total of (256 * 2 = 512) Glyphs/Controls
    # 256 in Alphanumeric Mode
    # 256 in Graphics Mode
  # Some unused/unmapped [Control] values (denoted "-x-")
  # Some ASCII   Controls  {BEL, BS, ...}
  # Some Bespoke Controls  {MODE_*, INK_*,, ...}
  # Some ASCII   Glyphs    {0-9, A-Z, ...}
  # Some Bespoke Glyphs    {Arrows, Fractions, ...}
  # A complete set of 64 {2x3} Graphical Glyphs

If you don't know why you might want it ...You don't want it.

(  EXAMPLE CODE  )
 `--------------'

This code is intended to be readable, and somewhat formatted to reflect the
tables [above].  It is certainly optimisable.

  # Control codes, ASCII-ish
  # Control codes, (mostly) NOT ASCII (exception is #127)
  # Glyphs (mostly ASCII) common to all four modes
  # Special cases
  # Alpha mode    (where lo==hi)
  # Graphics mode (where lo==hi)


// Control codes, ASCII-ish
if (INRANGE(ch, 0x00,0x1F))
{
	switch (ch) {
		case   0 :  return   FN_NOTHING;      //  NUL
		case   1 :  return FN_PRINTER_NEXT;   // (SOH)
		case   2 :  return FN_PRINTER_START;  //  STX
		case   3 :  return FN_PRINTER_STOP;   //  ETX
		case   4 :  return   FN_NOTHING;      // (EOT)
		case   5 :  return   FN_NOTHING;      // (ENQ)

		case   6 :  return FN_VDU_ENABLE;     //  ACK  <----.
		case   7 :  return FN_BEEP;           //  BEL       |

		case   8 :  return FN_CUR_BACK;       //  BS   --.  |
		case   9 :  return FN_CUR_FORWARD;    //  TAB  --|  |
		case  10 :  return FN_CUR_DOWN;       //  LF   --|  |
		case  11 :  return FN_CUR_UP;         // (VT ) --|  |

		case  12 :  return FN_CLS;            //  FF     |  |

		case  13 :  return FN_START_OF_LINE;  //  CR   --'  |

		case  14 :  return FN_MODE_PAGED;     // (SO )      |
		case  15 :  return FN_MODE_SCROLL;    // (SI )      |

		case  16 :  return   FN_NOTHING;      // (DLE)      |
		case  17 :  return   FN_NOTHING;      // (DC1)      |
		case  18 :  return   FN_NOTHING;      // (DC2)      |
		case  19 :  return   FN_NOTHING;      // (DC3)      |
		case  20 :  return   FN_NOTHING;      // (DC4)      |
		case  21 :  return FN_VDU_DISABLE;    //  NAK  <----'

		case  22 :  return FN_MODE_SELECT;    // (SYN)
		case  23 :  return FN_CHR_PROGRAM;    // (ETB)
		case  24 :  return   FN_NOTHING;      // (CAN)
		case  25 :  return   FN_NOTHING;      // (EM )
		case  26 :  return   FN_NOTHING;      // (SUB)
		case  27 :  return   FN_NOTHING;      // (ESC)
		case  28 :  return   FN_NOTHING;      // (FS )
		case  29 :  return   FN_NOTHING;      // (GS )
		case  30 :  return FN_CURSOR_HOME;    // (RS )
		case  31 :  return FN_CURSOR_MOVE;    // (US )
	}
}

// Control codes, (mostly) NOT ASCII (exception is #127)
else if (INRANGE(ch, 0x7F,0x9F))
{
	switch (ch) {
		case 127 :  return FN_DELETE;         //  DEL  [ASCII]
		case 128 :  return   FN_NOTHING;

		case 129 :  return FN_ALPHA_RED;      //
		case 130 :  return FN_ALPHA_GREEN;    //
		case 131 :  return FN_ALPHA_YELLOW;   //
		case 132 :  return FN_ALPHA_BLUE;     //
		case 133 :  return FN_ALPHA_MAGENTA;  //
		case 134 :  return FN_ALPHA_CYAN;     //
		case 135 :  return FN_ALPHA_WHITE;    // *Line Default

		case 136 :  return FN_ATTR_FLASH
		case 137 :  return FN_ATTR_STEADY     // *Line Default

		case 138 :  return   FN_NOTHING;
		case 139 :  return   FN_NOTHING;
		case 140 :  return FN_HEIGHT_NORMAL;  // *Line Default
		case 141 :  return FN_HEIGHT_DOUBLE;
		case 142 :  return   FN_NOTHING;
		case 143 :  return   FN_NOTHING;
		case 144 :  return   FN_NOTHING;

		case 145 :  return FN_INK_RED;
		case 146 :  return FN_INK_GREEN;
		case 147 :  return FN_INK_YELLOW;
		case 148 :  return FN_INK_BLUE;
		case 149 :  return FN_INK_MAGENTA;
		case 150 :  return FN_INK_CYAN;
		case 151 :  return FN_INK_WHITE;

		case 152 :  return FN_DISP_CONCEAL;

		case 153 :  return FN_GRAPH_CONTIG;   // *Line Default
		case 154 :  return FN_GRAPH_SPACED;
		case 155 :  return   FN_NOTHING;

		case 156 :  return FN_PAPER_BLACK;    // *Line Default
		case 157 :  return FN_PAPER_NEW;

		case 158 :  return FN_GRAPH_HOLD;
		case 159 :  return FN_GRAPH_RELEASE;  // *Line Default
	}
}

// Glyphs (mostly ASCII) common to all four modes
else if (INRANGE(ch, 0x40,0x5E) || INRANGE(ch, 0xC0,0xDE))
{
	switch (ch) {
		case 0x5B:  case 0xDB:  return GLYPH_LEFT;
		case 0x5C:  case 0xDC:  return GLYPH_HALF;
		case 0x5D:  case 0xDD:  return GLYPH_RIGHT;
		case 0x5E:  case 0xDE:  return GLYPH_UP;
		// {32, 35, 64, 65 .. 90) == {' ', '#', '@', 'A', ..., 'Z'}
		default:                return GLYPH_ASCII;
	}
}

// Special cases
else if (   ((ch & ~0x80) == 0x23)    // 0x23, 0xA3
         || ((ch & ~0x80) == 0x5F)    // 0x5F, 0xDF
         || ((ch & ~0x80) == 0x60)    // 0x60, 0xE0
         || ((ch & ~0x80) == 0x7F) )  // 0x7F, 0xFF
{
	switch (ch) {
		case 0x23:  return GLYPH_ASCII;  // "#" - hash
		case 0xDF:  return GLYPH_HASH;   // Also "#" - hash
		case 0x7F:  return FN_DELETE;
	}
	if        (mode == ALPHA) {
		switch (ch) {
			case 0xA3:  return GLYPH_GBP;
			case 0x60:  return GLYPH_GBP;
			case 0x5F:  return GLYPH_DASH;
			case 0xE0:  return GLYPH_DASH;
			case 0xFF:  return GLYPH_BLOCK;
		}
	} else if (mode == GRAPHICS) {
		switch (ch) {
			case 0xA3:  return GFX(0,0,3);
			case 0x60:  return GFX(0,0,3);
			case 0x5F:  return GFX(2,0,0);
			case 0xE0:  return GFX(2,0,0);
			case 0xFF:  return GFX(3,3,3);
		}
	}
}

// Alpha mode (where lo==hi)
else if (mode == ALPHA)
{
	switch (ch) {
		case 0x7B:  case 0xFB:  return GLYPH_1QUART;  // 1/4
		case 0x7C:  case 0xFC:  return GLYPH_DPIPE;   // ||
		case 0x7D:  case 0xFD:  return GLYPH_3QUART;  // 3/4
		case 0x7E:  case 0xFE:  return GLYPH_DIVIDE;  // Maths divide symbol
		default                 return GLYPH_ASCII;
	}
}

// Graphics mode (where lo==hi)
else if (mode == GRAPHICS)
{
	int  mask;

	if (INRANGE(ch, 0x20, 0x3F))  mask = 0x1F ;
	else                          mask = 0x3F ;

	int  btm = ((ch & mask) >> 4) & 0x03;
	int  mid = ((ch & mask) >> 2) & 0x03;
	int  top = ((ch & mask)     ) & 0x03;

	return GFX(btm, mid, top);
}

This script will convert the glyph diagrams in to usable C data.

There are many maanny ways to optimise this output.

------------- >-8 ------------- >-8 ------------- >-8 ------------- >-8 ------------- >-8 ------------- >-8 ------------- >-8 ------------- 
#!/bin/bash

##================================================================================================================
doit () {
	file=$1
	hdr=$2
	last=$3
	((last--))

	cnt=0  ## Array index
	n=0    ## Line counter

	while read -r l ; do         ## Read a line
		[[ -z $l ]] && continue  ## Ignore empty lines

		eval "bits${n}=(\$l)"      ## Get a full line of bits
		((++n < 10)) && continue   ## ...until we have 10 lines

		for ((col = 0;  col < 16;  col++)) ; do  ##   Max 16 glyphs per line
			[[ -z ${bits0[$col]} ]] && continue  ##     Ignore unused chars

			chr=$(case "$cnt" in                 ##     Find chr value
				("95")  echo 255 ;;              ##       ...
				("96")  echo 163 ;;              ##       ...
				("97")  echo 223 ;;              ##       ...
				("98")  echo 224 ;;              ##       ...
				(*   )  echo $((cnt + 32)) ;;    ##       ...
			esac)                                ##       ...
			printf -v chrx "0x%02X" $chr         ##     Convert to chr value hex

			## Start array for this glyph
			echo -en "\t\t{ // ---------------------[$cnt] -> chr($chr) = $chrx\n\t\t"
			for ((row = 0;  row < 10;  row++)) ; do            ## 10 rows
				eval "x=\${bits$row[$col]}"                    ## Get a row

				val=0                                          ## Calculate value
				for ((bit = 0;  bit < 6;  bit++)); do          ## 6 bits per line
					((val <<= 1))                              ##   Make way for new bit
					[[ "${x:bit:1}" != "." ]] && ((val |= 1))  ##   Add bit in (if present)
				done                                           ## .
				printf -v valx "0x%02X" $val                   ## Get hex value

				if ((row == 9)) ; then                         ## Last row of glyph?
					echo -en "  $valx  /* $x */\n\t\t}"        ##   No comma after value; close array
					((cnt != $last)) && echo -e ","            ##   NOT the last glyph? .. Append a comma
				else                                           ## Glyph row 0..8
					echo -en "  $valx, /* $x */\n\t\t"         ##   Comma after value; leave array open
				fi                                             ## .
			done
			((cnt++))                                          ## Next glyph
		done
		n=0
	done < <(sed -n "/^\*\*+${hdr}.*/,/^\*\*-${hdr}/p" ${file}  | sed '1d;$d')
}

##================================================================================================================
## ENTRY POINT
##
echo "uint8_t  glyph[2][(6 *16) +3][10] = {"

echo -e "\t{ // Alphanumeric Glyphs ==============================================="
doit $0 ALPHA $(((6 *16) +3))
echo -e "\n\t},"

echo -e "\t{ // Graphics Glyphs ==================================================="
doit $0 GRAPHIC $(((6 *16) +3))
echo -e "\n\t}"

echo "};"

exit 0
